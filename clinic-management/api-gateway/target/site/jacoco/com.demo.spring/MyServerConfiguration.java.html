package com.demo.spring;

import org.springframework.boot.context.properties.ConfigurationProperties;

@ConfigurationProperties
public class MyServerConfiguration {

	private String patientServer = "http://localhost:9191";
	private String clinicServer = "http://localhost:9193";
	private String appointmentServer = "http://localhost:9192";

	public String getPatientServer() {
		return patientServer;
	}

	public String getClinicServer() {
		return clinicServer;
	}

	public String getAppointmentServer() {
		return appointmentServer;
	}

}
